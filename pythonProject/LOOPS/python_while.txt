# While Loop
# Used to execute a block of code multiple times
# Useful for building interactive programs and games

# While statement - while condition:
# Define a variable called "i" (short for index) and set it to 1
i = 1

# Set condition to i is less than or equal to 5
while i <= 5:
    
    # Print i on the terminal
    print(i)

    # Increment i by 1
    # Otherwise, we will be stuck in an infinite loop
    # The condition will always be true. One is always less than 5
    i = i + 1

    # At some point, "i" is going to be 6 and this is when the condition will be false. And we will jump out of the loop

# Print statement
print("Done!")
